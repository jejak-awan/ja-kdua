<script setup lang="ts">
import { ref, computed } from 'vue';
import { useSocialShare } from '@/composables/useSocialShare';
 
const props = withDefaults(defineProps<{
  url?: string;
  title?: string;
  description?: string;
  hashtags?: string[];
  compact?: boolean;
  showCount?: boolean;
  shareCount?: number;
}>(), {
  url: () => window.location.href,
  title: 'Bagikan',
  description: '',
  hashtags: () => [],
  compact: false,
  showCount: false,
  shareCount: 0,
});
 
const emit = defineEmits<{
  (e: 'shared', payload: { platform: string; url: string }): void;
  (e: 'copied', url: string): void;
}>();
 
const { shareFacebook: _shareFacebook, shareTwitter: _shareTwitter, shareLinkedIn: _shareLinkedIn, shareWhatsApp: _shareWhatsApp, copyToClipboard } = useSocialShare();
const copied = ref(false);
 
const buttonClasses = computed(() => 
  props.compact 
    ? 'p-2 rounded-full flex items-center justify-center transition-colors hover:shadow-md'
    : 'px-4 py-2 rounded-lg flex items-center justify-center transition-colors hover:shadow-md'
);
 
const shareFacebook = () => {
  _shareFacebook(props.url, props.title);
  emit('shared', { platform: 'facebook', url: props.url });
};
 
const shareTwitter = () => {
  _shareTwitter(props.url, props.title, props.hashtags);
  emit('shared', { platform: 'twitter', url: props.url });
};
 
const shareLinkedIn = () => {
  _shareLinkedIn(props.url, props.title, props.description);
  emit('shared', { platform: 'linkedin', url: props.url });
};
 
const shareWhatsApp = () => {
  _shareWhatsApp(props.url, props.title);
  emit('shared', { platform: 'whatsapp', url: props.url });
};
 
const copyLink = async () => {
  const success = await copyToClipboard(props.url);
  if (success) {
    copied.value = true;
    emit('copied', props.url);
    setTimeout(() => {
      copied.value = false;
    }, 2000);
  }
};
</script>
